    new Init({
     discord: require('discord.js'),
     colors:{
         success: "#50CA2B",
         error: "#D31919"
     },
     messages:{
         error:{
             title: "Uh oh! An error appeared!",
             description: "Hey there! Either something went wrong with my robot code, or you did something wrong! Well, this is what happened! {str}"
         },
         success:{
